    \hypertarget{dataset-description}{%
\subsubsection{Dataset Description}\label{dataset-description}}

\begin{itemize}
\tightlist
\item
  mpg: Miles per gallon (Continuous variable)
\item
  cylinders: Power unit of an engine (Categorical variable)
\item
  displacement: Combined swept volume of the pistons inside the
  cylinders of an engine(Continuous variable)
\item
  horsepower: Power an engine produces(Continuous variable)
\item
  weight: mass of a vehicle (Continuous variable)
\item
  acceleration: the rate of change of the velocity of a car (Continuous
  variable)
\item
  year: the time taken after a car is made (Discrete variable)
\item
  origin: 1 is a car made in america, 2 in europe and 3 in asia or other
  part of the world (Categorical variable)
\end{itemize}
